155 inline constexpr double Clamp01(double x) noexcept
156 {
157 return (x < 0.0) ? 0.0 : (x > 1.0) ? 1.0 : x;
158 }
159
160 // A helper function to calculate linear interpolation
161 template <std::size_t N>
162 Color CalcLerp(double x, const Color (&data)[N])
163 {
164 const double a = Clamp01(x) * (N - 1);
165 const double i = std::floor(a);
166 const double t = a - i;
167 const Color& c0 = data[static_cast<std::size_t>(i)];
168 const Color& c1 = data[static_cast<std::size_t>(std::ceil(a))];
169
170 return (1.0 - t) * c0 + t * c1;
171 }
172
173 inline double QuantizeArgument(double x, unsigned int num_levels)
174 {
175 // Clamp num_classes to range [1, 255].
176 num_levels = (std::max)(1u, (std::min)(num_levels, 255u));
177
178 const double interval_length = 255.0 / num_levels;
179
180 // Calculate index of the interval to which the given x belongs to.
181 // Substracting eps prevents getting out of bounds index.
182 constexpr double eps = 0.0005;
183 const unsigned int index = static_cast<unsigned int>((x * 255.0 - eps) / interval_length);
184
185 // Calculate upper and lower bounds of the given interval.
186 const unsigned int upper_boundary = static_cast<unsigned int>(index * interval_length + interval_length);
187 const unsigned int lower_boundary = static_cast<unsigned int>(upper_boundary - interval_length);
188
189 // Get middle "coordinate" of the given interval and move it back to [0.0, 1.0] interval.
190 const double xx = static_cast<double>(upper_boundary + lower_boundary) * 0.5 / 255.0;
191
192 return xx;
193 }
194 }

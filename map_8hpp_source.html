195 getMaxData(const Eigen::Vector3f& point_W, const int scale_desired) const
196 {
197 Eigen::Vector3i voxel_coord;
198
199 if constexpr (SafeB == Safe::Off) // Evaluate at compile time
200 {
201 pointToVoxel<Safe::Off>(point_W, voxel_coord);
202 }
203 else {
204 if (!pointToVoxel<Safe::On>(point_W, voxel_coord)) {
205 return DataType();
206 }
207 }
208
209 return se::visitor::getMaxData(octree_, voxel_coord, scale_desired);
210 }
